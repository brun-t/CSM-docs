195Arena *create_arena(size_t capacity) {
196 Arena *arena = (Arena *)malloc(sizeof(Arena));
197
198 if (arena == NULL) {
199 return NULL;
200 }
201
202 arena->capacity = capacity;
203 arena->actual_size = 0;
204 arena->block = (uint8_t *)malloc(arena->capacity);
205
206 if (arena->block == NULL) {
207 free(arena);
208 return NULL;
209 }
210
211 return arena;
212}

214Arena_ptr arena_alloc(Arena *arena, size_t size) {
215 if (arena == NULL || size == 0)
216 return (Arena_ptr){.size = 0, .block = NULL};
217
218 if (arena->actual_size + size > arena->capacity)
219 return (Arena_ptr){.size = 0, .block = NULL};
220
221 uint8_t *block = &arena->block[arena->actual_size];
222 arena->actual_size += size;
223
224 return (Arena_ptr){.size = size, .block = block};
225}

227bool arena_realloc(Arena *arena, size_t extra_capacity) {
228 void *ptr = realloc(arena->block, arena->actual_size + extra_capacity);
229 if (ptr == NULL)
230 return false;
231 arena->block = (uint8_t *)ptr;
232 arena->capacity += extra_capacity;
233 return true;
234}

241Ptr_stack *create_stack(size_t capacity) {
242 Ptr_stack *ptr_stack = (Ptr_stack *)malloc(sizeof(Ptr_stack));
243 if (ptr_stack == NULL)
244 return NULL;
245
246 Dyn_ptr *dyn_ptrs = (Dyn_ptr *)malloc(capacity * sizeof(Dyn_ptr));
247 if (dyn_ptrs == NULL) {
248 free(ptr_stack);
249 return NULL;

252 ptr_stack->capacity = capacity; // Initialize capacity

254 if (arena == NULL) {
255 free(dyn_ptrs);
256 free(ptr_stack);

258 }

260 ptr_stack->length = 0;
261 ptr_stack->ptr_list = dyn_ptrs;
262 ptr_stack->arena = arena;

267Dyn_ptr *stack_new_ptr(Ptr_stack *stack, void *data, size_t dataSize) {
268 size_t growth = dataSize * (size_t)2;
269 if (growth < 1024) { // minimum of 1KB growth.
270 growth = 1024;
271 }
272 if (stack->length >= stack->capacity && !arena_realloc(stack->arena, growth)) {
273 return NULL;
274 }
275
276 Arena_ptr arena_ptr = arena_alloc(stack->arena, sizeof(Dyn_ptr));
277 if (arena_ptr.block == NULL) {
278 return NULL;
279 }
280
281 Dyn_ptr *dyn_ptr = (Dyn_ptr *)arena_ptr.block;
282 dyn_ptr->ptr = NULL;
283
284 stack->ptr_list[stack->length] = *dyn_ptr;
285 stack->length++;
286
287 dyn_ptr = &stack->ptr_list[stack->length - 1];
288 dyn_ptr->size = dataSize;
289 dyn_ptr->dealloc = null_deallocator;
290 dyn_ptr_alloc(stack, dyn_ptr, data, dataSize);
291 if (dyn_ptr->ptr == NULL)
292 return NULL;
293 return dyn_ptr;
294}

296void dyn_ptr_alloc(Ptr_stack *stack, Dyn_ptr *dyn_ptr, void *data,
297 size_t size) {
298 if (stack == NULL || dyn_ptr == NULL || data == NULL || size == 0) {
299 return;
300 }
301
302 Arena_ptr data_arena_ptr = arena_alloc(stack->arena, size);
303
304 if (data_arena_ptr.block == NULL) {
305 return;
306 }
307
308 memcpy(data_arena_ptr.block, data, size);
309
310 dyn_ptr->ptr = data_arena_ptr.block;
311 dyn_ptr->size = size;
312}
